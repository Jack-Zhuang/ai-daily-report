## 💡 核心方法

### 方法概述

ICEBREAKER将人类破冰行为形式化为两阶段握手：先从用户特征中提取共鸣触发点，再生成个性化开场白。


### 详细设计

**阶段一：共鸣感知兴趣提取（Resonance-Aware Interest Distillation）** 采用多信号融合策略，聚合用户画像、历史交互、社交关系等多维度信息。具体通过温度调节的参数β控制兴趣分布的锐度，使得高频兴趣更加突出。论文引入h⁺和h_u的区分策略，前者代表高置信度兴趣信号，后者代表待验证的不确定信号，形成正负样本的协同训练。投影层Proj(·)和编码器E(·)将多模态信息映射到统一语义空间。

**阶段二：个性化开场白生成** 基于提取的共鸣触发点，结合对话上下文模板，生成符合用户画像的开场白。论文引入效用-多样性权衡参数λ，确保开场白既具有实际价值（引导有效对话），又保持足够的多样性（避免重复机械）。

两阶段通过端到端优化实现，损失函数同时考虑兴趣提取的准确性和开场白的生成质量。

import torch
import torch.nn as nn

# ============================================================
# 阶段一：共鸣感知兴趣提取 (Resonance-Aware Interest Distillation)
# ============================================================

class ResonanceAwareInterestExtractor(nn.Module):
    """多信号融合的兴趣提取器"""
    
    def __init__(self, embed_dim=128, beta_init=1.0):
        super().__init__()
        # 多模态编码器
        self.encoder = nn.Embedding(10000, embed_dim)  # 用户/物品/社交编码
        
        # 温度调节参数 (控制兴趣分布锐度)
        self.beta = nn.Parameter(torch.tensor(beta_init))
        
        # 投影层 (映射到统一语义空间)
        self.projection = nn.Sequential(
            nn.Linear(embed_dim, embed_dim),
            nn.ReLU(),
            nn.Linear(embed_dim, embed_dim)
        )
        
    def forward(self, user_profile, history, social_net):
        """
        输入: 用户画像、历史交互、社交关系
        输出: h+ (高置信度兴趣), h- (待验证信号)
        """
        # Step 1: 多信号融合 - 聚合多维度信息
        fused = self.encoder(user_profile) + \
                self.encoder(history) * 0.5 + \
                self.encoder(social_net) * 0.3
        
        # Step 2: 温度调节 - 通过β控制兴趣分布锐度
        interest_logits = self.projection(fused) / self.beta
        
        # Step 3: 区分策略 - 高置信度 vs 待验证
        confidence_scores = torch.softmax(interest_logits, dim=-1)
        
        # h+: 高置信度兴趣信号 (top-k)
        h_plus, _ = torch.topk(confidence_scores, k=5, dim=-1)
        
        # h-: 待验证的不确定信号 (low confidence)
        h_minus = 1.0 - confidence_scores
        
        return h_plus, h_minus, confidence_scores


# ============================================================
# 阶段二：个性化开场白生成
# ============================================================

class OpeningStatementGenerator(nn.Module):
    """基于共鸣触发点生成开场白"""
    
    def __init__(self, embed_dim=128, vocab_size=10000, lambda_diversity=0.3):
        super().__init__()
        # 效用-多样性权衡参数
        self.lambda_div = lambda_diversity
        
        # 上下文编码器
        self.ctx_encoder = nn.LSTM(embed_dim, embed_dim, batch_first=True)
        
        # 开场白解码器
        self.decoder = nn.GRU(embed_dim, embed_dim, batch_first=True)
        self.output_proj = nn.Linear(embed_dim, vocab_size)
        
    def forward(self, interest_signal, dialogue_ctx):
        """
        输入: 兴趣信号、对话上下文
        输出: 开场白序列
        """
        # Step 1: 编码对话上下文
        ctx_encoded, _ = self.ctx_encoder(dialogue_ctx)
        ctx_hidden = ctx_encoded[:, -1]  # 取最后隐状态
        
        # Step 2: 融合共鸣触发点
        combined = interest_signal + ctx_hidden
        
        # Step 3: 生成开场白 (简化示例)
        batch_size = combined.size(0)
        hidden = combined.unsqueeze(0)
        outputs = []
        
        for t in range(20):  # 最大长度20
            output, hidden = self.decoder(hidden)
            logits = self.output_proj(output.squeeze(0))
            outputs.append(logits)
            hidden = hidden  # 自回归
            
        generated = torch.stack(outputs, dim=1)
        return generated
    
    def compute_loss(self, pred_logits, target, interest_loss):
        """端到端联合损失"""
        # 生成损失 (效用)
        utility_loss = nn.CrossEntropyLoss()(pred_logits.view(-1, pred_logits.size(-1)), 
                                              target.view(-1))
        
        # 多样性惩罚 (避免重复机械)
        diversity_penalty = -self.lambda_div * torch.var(pred_logits, dim=-1).mean()
        
        # 总损失 = 兴趣提取损失 + 生成效用损失 + 多样性奖励
        total_loss = interest_loss + utility_loss + diversity_penalty
        
        return total_loss


# ============================================================
# 两阶段端到端模型
# ============================================================

class TwoStageConversationStarter(nn.Module):
    """端到端两阶段对话开场系统"""
    
    def __init__(self):
        super().__init__()
        self.interest_extractor = ResonanceAwareInterestExtractor()
        self.opening_generator = OpeningStatementGenerator()
        
    def forward(self, user_profile, history, social_net, dialogue_ctx, target):
        # ---------- Stage 1: 兴趣提取 ----------
        h_plus, h_minus, scores = self.interest_extractor(
            user_profile, history, social_net
        )
        
        # 正负样本协同训练损失 (简化)
        interest_loss = -torch.log(h_plus.mean() + 1e-8) - \
                       torch.log(h_minus.mean() + 1e-8)
        
        # ---------- Stage 2: 开场白生成 ----------
        combined_interest = h_plus  # 使用高置信度兴趣
        generated = self.opening_generator(combined_interest, dialogue_ctx)
        
        # ---------- 端到端联合优化 ----------
        total_loss = self.opening_generator.compute_loss(
            generated, target, interest_loss
        )
        
        return generated, total_loss


# ============================================================
# 使用示例
# ============================================================

if __name__ == "__main__":
    # 初始化模型
    model = TwoStageConversationStarter()
    
    # 模拟输入数据
    batch_size = 4
    user_profile = torch.randint(0, 10000, (batch_size, 20))
    history = torch.randint(0, 10000, (batch_size, 50))
    social_net = torch.randint(0, 10000, (batch_size, 30))
    dialogue_ctx = torch.randint(0, 10000, (batch_size, 10, 128))  # seq_len=10
    target = torch.randint(0, 10000, (batch_size, 20))
    
    # 前向传播
    generated, loss = model(user_profile, history, social_net, dialogue_ctx, target)
    
    print(f"生成的句子形状: {generated.shape}")  # [batch, seq_len, vocab]
    print(f"总损失: {loss.item():.4f}")
    
    # 端到端训练
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    optimizer.zero_grad()
    loss.backward()
    optimizer.step()
    
    print("✓ 两阶段端到端训练完成!")
